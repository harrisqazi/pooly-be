async function runTest(testNumber, testName, testFn) {
  log(`\n=== TEST ${testNumber}: ${testName} ===`)
  let attempts = 0
  while (attempts < 3) {
    try {
      const result = await testFn()
      if (result.pass) {
        log(`✅ PASS: ${testName}`)
        log(`Response: ${JSON.stringify(result.data, null, 2)}`)
        await createCheckpoint(`test-${testNumber}-passed`)
        return result
      } else {
        attempts++
        log(`❌ FAIL attempt ${attempts}: ${result.reason}`)
        log(`Response was: ${JSON.stringify(result.data, null, 2)}`)
        log(`Diagnosing...`)
        await result.diagnoseAndFix()
        if (attempts >= 3) {
          log(`🛑 COULD NOT AUTO-FIX after 3 attempts`)
          log(`Last error: ${result.reason}`)
          log(`Possible causes:`)
          result.possibleCauses.forEach(c => log(`  - ${c}`))
          await writeLog()
          await createCheckpoint(`test-${testNumber}-failed-stopping`)
          process.exit(1)
        }
        log(`Retrying test ${testNumber}...`)
      }
    } catch (err) {
      attempts++
      log(`💥 Exception: ${err.message}`)
      if (attempts >= 3) {
        log(`🛑 STOPPING - unrecoverable error in test ${testNumber}`)
        await writeLog()
        process.exit(1)
      }
    }
  }
}
