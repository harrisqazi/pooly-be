STEP 4 — CREATE ROUTES/AGENT.JS (NEW FILE)
============================================================
Create routes/agent.js from scratch.
Use routes/topups.js as the template for import style and structure.

Required imports:
  const express = require('express');
  const router = express.Router();
  const crypto = require('crypto');
  const jwt = require('jsonwebtoken');
  const { supabase } = require('../config/providers');
  const idempotencyMiddleware = require('../middleware/idempotency');
  const { createLedgerEntry, updateGroupBalance } = require('../utils/ledger');

HELPER: verifyAdminKey(req, res, next)
  Check: req.headers.authorization === 'Bearer ' + process.env.ADMIN_KEY
  If missing or wrong: return res.status(401).json({ error: 'Invalid admin key' })
  Otherwise: next()

HELPER: computeRulesHash(rulesObj)
  Sort keys alphabetically, JSON.stringify, then SHA256 hex digest.
  Use crypto.createHash('sha256').update(JSON.stringify(sorted)).digest('hex')

HELPER: computeTokenHash(token)
  crypto.createHash('sha256').update(token).digest('hex')

IN-MEMORY RATE TRACKER
  const rateMap = new Map();
  Key: group_id, Value: { count: number, resetAt: timestamp }
  Reset when Date.now() > resetAt (60 second windows)
  Flag (do not block) if count > 10 in the window

MIDDLEWARE: verifyAgentToken(req, res, next)
  - Extract Bearer token from Authorization header
  - jwt.verify with process.env.AGENT_JWT_SECRET
  - If expired or invalid: 401
  - computeTokenHash, query agent_tokens:
      not found → 401 'Token not registered'
      revoked=true → 401 'Token revoked'
      expires_at < now → 401 'Token expired'
  - Attach: req.agentGroupId, req.agentRulesHash, req.agentTokenHash
  - next()

MIDDLEWARE: checkAgentRules(req, res, next)
  - Fetch agent_rules WHERE group_id = req.agentGroupId
  - If none: 403 'No rules configured for this group'
  - amount_cents = Math.round(req.body.amount * 100)

  CHECK 1 — per transaction cap:
    if max_per_txn set AND amount_cents > max_per_txn:
      insert agent_spend_log (status='blocked', block_reason='per_txn_cap', amount=amount_cents)
      insert anomaly_log (event_type='rule_breach', group_id, payload={reason:'per_txn_cap'})
      return 403 { error: 'Exceeds per-transaction cap', cap: max_per_txn/100 }

  CHECK 2 — daily cap:
    SELECT COALESCE(SUM(amount),0) FROM agent_spend_log
    WHERE group_id=X AND status='approved'
    AND created_at >= start of today UTC (use: new Date().setUTCHours(0,0,0,0))
    if daily_spend_cap set AND (todaySum + amount_cents) > daily_spend_cap:
      insert agent_spend_log (status='blocked', block_reason='daily_cap')
      insert anomaly_log (event_type='rule_breach', payload={reason:'daily_cap'})
      return 403 { error: 'Daily spend cap reached', cap: daily_spend_cap/100 }

  CHECK 3 — MCC allowlist:
    if allowlist_mcc set AND allowlist_mcc.length > 0
    AND req.body.mcc provided AND NOT in allowlist_mcc:
      insert agent_spend_log (status='blocked', block_reason='mcc_blocked')
      return 403 { error: 'MCC not in allowlist' }

  attach req.agentRules = rules row
  next()

MIDDLEWARE: anomalyDetection(req, res, next)
  RATE CHECK:
    get/update rateMap for req.agentGroupId
    if count > 10 in 60s window:
      req.anomalyFlag = true
      req.anomalyReason = 'rate_spike'
      insert anomaly_log (event_type='rate_spike', group_id, source_ip=req.ip)
    (do NOT block)

  IP CHECK:
    query anomaly_log WHERE group_id=X AND created_at > 7 days ago
    extract all source_ip values from results
    if req.ip not in those values:
      req.anomalyFlag = req.anomalyFlag || true
      insert anomaly_log (event_type='new_ip', group_id, source_ip=req.ip, payload={ip:req.ip})
    (do NOT block)

  next()

POST /api/agent/rules
  Middleware: verifyAdminKey
  Body: { group_id, daily_spend_cap, max_per_txn, allowlist_mcc, quorum_required }
  - daily_spend_cap and max_per_txn come in as DOLLARS — store as CENTS (*100)
  - computeRulesHash on the cents values object
  - Upsert into agent_rules ON CONFLICT (group_id) DO UPDATE
  - Return: { rule_id, group_id, rules_hash, daily_spend_cap_cents, max_per_txn_cents }

POST /api/agent/token
  Middleware: verifyAdminKey
  Body: { group_id }
  - Fetch agent_rules for group_id — 404 if none
  - jwt.sign({ group_id, rules_hash, type:'agent' }, process.env.AGENT_JWT_SECRET, { expiresIn:'1h' })
  - computeTokenHash
  - expires_at = new Date(Date.now() + 3600000).toISOString()
  - Insert into agent_tokens: { group_id, token_hash, rules_hash, expires_at }
  - Return: { token, expires_at }
  - NEVER log the raw token value

POST /api/agent/pay
  Middleware chain (in order):
    idempotencyMiddleware, verifyAgentToken, checkAgentRules, anomalyDetection
  Body: { amount, merchant_name, mcc, memo, idempotency_key }

  Flow:
  1. amount_cents = Math.round(req.body.amount * 100)
  2. Fetch group from groups WHERE id = req.agentGroupId
  3. If group.total_balance < amount_cents: 402 { error: 'Insufficient balance' }
  4. Determine provider:
       if process.env.PROVIDER_STRIPE === 'true' → stripe
       else if process.env.PROVIDER_ISSUING === 'lithic' → lithic
       else if process.env.PROVIDER_BANK_RAILS === 'modern_treasury' → modern_treasury
       else → paytheory

  5. Provider routing:
     lithic:
       fetch group.card_token — if missing: 400 'Group has no Lithic card'
       const { lithic } = require('../config/providers')
       simulate authorization (lithic.cards.retrieve(group.card_token))
       provider_ref = group.card_token + '_agent_' + Date.now()

     modern_treasury:
       const { modernTreasuryClient } = require('../config/providers')
       POST /payment_orders: { amount: amount_cents, direction:'debit',
         currency:'USD', description: memo || 'Agent payment' }
       provider_ref = response.data.id

     paytheory:
       const { payTheoryClient } = require('../config/providers')
       POST /charges: { amount: amount_cents, description: memo || 'Agent payment' }
       provider_ref = response.data.id

     stripe:
       const stripe = require('stripe')(process.env.STRIPE_SECRET_KEY)
       stripe.paymentIntents.create({
         amount: amount_cents, currency: 'usd',
         metadata: { group_id: req.agentGroupId, memo, token_hash: req.agentTokenHash }
       })
       provider_ref = paymentIntent.id

  6. Insert into transactions:
       group_id: req.agentGroupId
       user_id: '00000000-0000-0000-0000-000000000000'
       type: 'agent_payment'
       amount: amount_cents
       description: memo || 'Agent payment'
       status: 'completed'
       merchant_name: req.body.merchant_name || null
       mcc: req.body.mcc || null
       payment_method: provider used

  7. createLedgerEntry({
       transaction_id: transaction.id,
       debit_account: 'expenses',
       credit_account: 'cash',
       amount: amount_cents,
       description: memo || 'Agent payment'
     })

  8. updateGroupBalance(req.agentGroupId)

  9. Insert into agent_spend_log:
       group_id: req.agentGroupId
       token_hash: req.agentTokenHash
       amount: amount_cents
       merchant_name, mcc, memo
       status: 'approved'
       provider: provider used
       anomaly_flag: req.anomalyFlag || false

  10. Return 200: {
        approved: true,
        amount_dollars: amount_cents / 100,
        provider,
        provider_ref,
        transaction_id: transaction.id,
        anomaly_flagged: req.anomalyFlag || false
      }

POST /api/agent/spend-log
  Middleware: verifyAdminKey
  Body: { group_id, limit=50, offset=0 }
  - Query agent_spend_log WHERE group_id ORDER BY created_at DESC
  - Map amount: divide by 100 for display
  - Return: { data, total_count, limit, offset }

POST /api/agent/revoke
  Middleware: verifyAdminKey
  Body: { token_hash }
  - UPDATE agent_tokens SET revoked=true WHERE token_hash=token_hash
  - Return: { revoked: true }

module.exports = router;
